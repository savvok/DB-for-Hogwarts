CREATE TABLE s308491.Crystal(Crystal_id SERIAL PRIMARY KEY, Kind VARCHAR(50) NOT NULL, Programming_type VARCHAR(50) NOT NULL);
CREATE TABLE s308491.Weapon(Weapon_id SERIAL PRIMARY KEY, Kind VARCHAR(50) NOT NULL, Description TEXT NOT NULL, Crystal_id INTEGER REFERENCES s308491.Crystal NOT NULL);
CREATE TABLE s308491.Habitat(Habitat_id SERIAL PRIMARY KEY, Location VARCHAR(20) NOT NULL, Climate TEXT NOT NULL);
CREATE TABLE s308491.Pithecanthrop(Pithecanthrop_id SERIAL PRIMARY KEY, Name VARCHAR(50) NOT NULL, Habitat_id INTEGER REFERENCES s308491.Habitat NOT NULL);
CREATE TABLE s308491.Belong(Weapon_id INTEGER REFERENCES s308491.Weapon NOT NULL, Pithecanthrop_id INTEGER REFERENCES s308491.Pithecanthrop NOT NULL, CONSTRAINT Belong_pkey PRIMARY KEY (Weapon_id, Pithecanthrop_id));
CREATE TABLE s308491.Animals(Animal_id SERIAL PRIMARY KEY, Food_type VARCHAR(10) NOT NULL, Kind VARCHAR(50) NOT NULL, Habitat_id INTEGER REFERENCES s308491.Habitat NOT NULL);
CREATE TABLE s308491.Protection_mean(Protection_id SERIAL PRIMARY KEY, Description TEXT NOT NULL);
CREATE TABLE s308491.Possess(Animal_id INTEGER REFERENCES s308491.Animals NOT NULL, Protection_id INTEGER REFERENCES s308491.Protection_mean NOT NULL, CONSTRAINT Possess_pkey PRIMARY KEY (Animal_id,Protection_id));
CREATE TABLE s308491.Hunt(Pithecanthrop_id INTEGER REFERENCES s308491.Pithecanthrop NOT NULL, Animal_id INTEGER REFERENCES s308491.Animals NOT NULL, Booty VARCHAR(50), CONSTRAINT Hunt_pkey PRIMARY KEY(Pithecanthrop_id, Animal_id));
